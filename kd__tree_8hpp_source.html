224 [[nodiscard]] static inline pcr::dist_t
225 get_dist_squared(const pcr::point_t &p1, const pcr::point_t &p2) {
226 pcr::dist_t x_diff = p1.x() - p2.x();

228 pcr::dist_t z_diff = p1.z() - p2.z();

230 pcr::dist_t dist_squared =
231 (x_diff * x_diff) + (y_diff * y_diff) + (z_diff * z_diff);
232 return dist_squared;

262 asm("# end get dist squared");
